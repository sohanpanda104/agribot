  ```
  VCC        ->   3.3V (Pin 1)
  GND        ->   GND (Pin 6)
  SDA        ->   SDA1 (GPIO 2, Pin 3)
  SCL        ->   SCL1 (GPIO 3, Pin 5)
  ```

### âœ… **Python Code using `mpu9250-jmdev` Library**

```python
from mpu9250_jmdev.registers import *
from mpu9250_jmdev.mpu_9250 import MPU9250
import time

# Initialize the MPU9250
mpu = MPU9250(
    address_ak=AK8963_ADDRESS,       # Magnetometer
    address_mpu_master=MPU9050_ADDRESS_68,  # Gyro + Accel
    address_mpu_slave=None,
    bus=1,
    gfs=GFS_250,      # Gyroscope full scale range
    afs=AFS_2G,       # Accelerometer full scale range
    mfs=AK8963_BIT_16,  # Magnetometer resolution
    mode=AK8963_MODE_C100HZ)  # Magnetometer data rate

mpu.calibrate()  # Optional
mpu.configure()

# Read sensor data
while True:
    accel = mpu.readAccelerometerMaster()
    gyro = mpu.readGyroscopeMaster()
    mag = mpu.readMagnetometerMaster()
    temp = mpu.readTemperatureMaster()

    print(f"Accel: {accel}")
    print(f"Gyro : {gyro}")
    print(f"Mag  : {mag}")
    print(f"Temp : {temp:.2f} Â°C")
    print("-" * 30)
    time.sleep(1)
```
